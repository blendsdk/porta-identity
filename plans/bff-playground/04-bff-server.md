## Overview

The core BFF application — a Koa web server that uses `openid-client` v6 to handle OIDC Authorization Code flow server-side. This document covers the project structure, package setup, server configuration, session management, and OIDC client integration.

### Request Flow

```
1. GET /auth/login?scenario=normalLogin
   └── Read scenario config → resolve org → build authorization URL
       └── openid-client: discovery(issuer) → authorizationCodeGrant.redirect()
           └── 302 → Porta authorization endpoint

2. User authenticates at Porta (login, 2FA, consent)
   └── 302 → http://localhost:4001/auth/callback?code=...&state=...

3. GET /auth/callback?code=...&state=...
   └── openid-client: authorizationCodeGrant(currentUrl, checks)
       └── Exchanges code for tokens server-side (client_secret_post + PKCE)
           └── Stores tokens in session → 302 → / (dashboard)

4. GET / (dashboard)
   └── Read tokens from session → decode JWTs → render dashboard template

5. POST /api/me
   └── openid-client: fetchUserInfo(accessToken)
       └── Returns JSON to page via AJAX

6. POST /api/refresh
   └── openid-client: refreshTokenGrant(refreshToken)
       └── Updates session with new tokens → returns JSON

7. POST /auth/logout
   └── Destroy session → redirect to Porta end_session_endpoint
```

### OIDC Client Setup (`src/oidc.ts`)

Uses `openid-client` v6 API:

```typescript
import * as client from 'openid-client';
import type { OrgConfig, M2MConfig } from './config.js';

// Cache discovered configurations per issuer
const discoveredConfigs = new Map<string, client.Configuration>();

/**
 * Discover and configure an openid-client for a specific org.
 * Caches the result so subsequent calls for the same org reuse discovery.
 */
export async function getOidcConfig(
  portaUrl: string,
  org: OrgConfig
): Promise<client.Configuration> {
  const issuer = `${portaUrl}/${org.slug}`;
  
  if (discoveredConfigs.has(issuer)) {
    return discoveredConfigs.get(issuer)!;
  }

  const config = await client.discovery(
    new URL(issuer),
    org.clientId,
    org.clientSecret,
    client.ClientSecretPost(org.clientSecret)
  );

  discoveredConfigs.set(issuer, config);
  return config;
}

/**
 * Build authorization URL with PKCE.
 * Returns the URL and the code_verifier + state + nonce for session storage.
 */
export function buildAuthorizationUrl(
  config: client.Configuration,
  redirectUri: string,
  scope: string
): { url: URL; code_verifier: string; state: string; nonce: string } {
  const code_verifier = client.randomPKCECodeVerifier();
  const code_challenge = await client.calculatePKCECodeChallenge(code_verifier);
  const state = client.randomState();
  const nonce = client.randomNonce();

  const parameters: Record<string, string> = {
    redirect_uri: redirectUri,
    scope,
    code_challenge,
    code_challenge_method: 'S256',
    state,
    nonce,
  };

  const url = client.buildAuthorizationUrl(config, parameters);
  return { url, code_verifier, state, nonce };
}

/**
 * Exchange authorization code for tokens.
 */
export async function exchangeCode(
  config: client.Configuration,
  currentUrl: URL,
  expectedState: string,
  expectedNonce: string,
  code_verifier: string
): Promise<client.TokenEndpointResponse> {
  return client.authorizationCodeGrant(config, currentUrl, {
    pkceCodeVerifier: code_verifier,
    expectedState,
    expectedNonce,
    idTokenExpected: true,
  });
}

/**
 * Refresh tokens using refresh_token grant.
 */
export async function refreshTokens(
  config: client.Configuration,
  refreshToken: string
): Promise<client.TokenEndpointResponse> {
  return client.refreshTokenGrant(config, refreshToken);
}

/**
 * Fetch UserInfo from the provider.
 */
export async function fetchUserInfo(
  config: client.Configuration,
  accessToken: string
): Promise<client.UserInfoResponse> {
  return client.fetchUserInfo(config, accessToken, 'sub');
}

/**
 * Introspect a token.
 */
export async function introspectToken(
  config: client.Configuration,
  token: string
): Promise<client.IntrospectionResponse> {
  return client.introspectionRequest(config, token);
}

/**
 * Get the end_session_endpoint URL for logout.
 */
export function getEndSessionUrl(
  config: client.Configuration,
  idToken: string,
  postLogoutRedirectUri: string
): URL {
  return client.buildEndSessionUrl(config, {
    id_token_hint: idToken,
    post_logout_redirect_uri: postLogoutRedirectUri,
  });
}

/**
 * Client credentials grant (M2M).
 */
export async function clientCredentialsGrant(
  portaUrl: string,
  m2m: M2MConfig
): Promise<client.TokenEndpointResponse> {
  const issuer = `${portaUrl}/${m2m.orgSlug}`;
  
  let config = discoveredConfigs.get(`m2m:${issuer}`);
  if (!config) {
    config = await client.discovery(
      new URL(issuer),
      m2m.clientId,
      m2m.clientSecret,
      client.ClientSecretPost(m2m.clientSecret)
    );
    discoveredConfigs.set(`m2m:${issuer}`, config);
  }

  return client.clientCredentialsGrant(config);
}
```

> **Note:** The exact `openid-client` v6 API may differ slightly — the implementation should consult the latest v6 README/examples at build time. The above represents the conceptual approach.

### Session Configuration (`src/session.ts`)

```typescript
import session from 'koa-session';
import Redis from 'ioredis';
import type Koa from 'koa';

export function configureSession(app: Koa, redisHost: string, redisPort: number) {
  const redis = new Redis({ host: redisHost, port: redisPort });
  
  const store = {
    async get(key: string) {
      const data = await redis.get(`bff:sess:${key}`);
      return data ? JSON.parse(data) : null;
    },
    async set(key: string, sess: unknown, maxAge?: number) {
      const ttl = maxAge ? Math.ceil(maxAge / 1000) : 86400; // default 24h
      await redis.set(`bff:sess:${key}`, JSON.stringify(sess), 'EX', ttl);
    },
    async destroy(key: string) {
      await redis.del(`bff:sess:${key}`);
    },
  };

  app.keys = ['bff-playground-secret-key']; // signing key for session cookie

  app.use(session({
    key: 'bff_session',     // cookie name — distinct from Porta's cookies
    maxAge: 86400000,       // 24 hours
    httpOnly: true,
    signed: true,
    store,
  }, app));
}
```

### Server (`src/server.ts`)

```typescript
import Koa from 'koa';
import Router from '@koa/router';
import serve from 'koa-static';
import bodyParser from 'koa-bodyparser';
import { loadConfig } from './config.js';
import { configureSession } from './session.js';
import { createAuthRoutes } from './routes/auth.js';
import { createDashboardRoutes } from './routes/dashboard.js';
import { createApiRoutes } from './routes/api.js';
import { createM2mRoutes } from './routes/m2m.js';
import { createHealthRoutes } from './routes/health.js';

const config = loadConfig();
const app = new Koa();
const router = new Router();

// Session
configureSession(app, config.redis.host, config.redis.port);

// Body parsing
app.use(bodyParser());

// Static files
app.use(serve('public'));

// Routes
createHealthRoutes(router);
createAuthRoutes(router, config);
createDashboardRoutes(router, config);
createApiRoutes(router, config);
createM2mRoutes(router, config);

app.use(router.routes());
app.use(router.allowedMethods());

// Start
const PORT = 4001;
app.listen(PORT, () => {
  console.log(`BFF Playground running on http://localhost:${PORT}`);
  console.log(`Porta: ${config.portaUrl}`);
  console.log(`Organizations: ${Object.keys(config.organizations).join(', ')}`);
});
```

### Auth Routes (`src/routes/auth.ts`)

```typescript
import Router from '@koa/router';
import type { BffConfig } from '../config.js';
import { getOidcConfig, buildAuthorizationUrl, exchangeCode, getEndSessionUrl } from '../oidc.js';

export function createAuthRoutes(router: Router, config: BffConfig) {
  
  // GET /auth/login?scenario=normalLogin (or ?org=no2fa)
  router.get('/auth/login', async (ctx) => {
    const scenarioKey = ctx.query.scenario as string;
    const orgKey = ctx.query.org as string || 'no2fa';
    
    // Resolve org from scenario or direct org param
    const scenario = scenarioKey ? config.scenarios[scenarioKey] : null;
    const resolvedOrgKey = scenario?.orgKey || orgKey;
    const org = config.organizations[resolvedOrgKey];
    
    if (!org) {
      ctx.status = 400;
      ctx.body = `Unknown organization: ${resolvedOrgKey}`;
      return;
    }

    const oidcConfig = await getOidcConfig(config.portaUrl, org);
    const { url, code_verifier, state, nonce } = buildAuthorizationUrl(
      oidcConfig,
      `${config.bffUrl}/auth/callback`,
      'openid profile email offline_access'
    );

    // Store PKCE + state in session
    ctx.session!.oidc = {
      code_verifier,
      state,
      nonce,
      orgKey: resolvedOrgKey,
      scenarioKey,
    };

    ctx.redirect(url.toString());
  });

  // GET /auth/callback
  router.get('/auth/callback', async (ctx) => {
    const { code_verifier, state, nonce, orgKey } = ctx.session!.oidc || {};
    
    if (!code_verifier || !state || !orgKey) {
      ctx.status = 400;
      ctx.body = 'No pending authentication flow';
      return;
    }

    const org = config.organizations[orgKey];
    const oidcConfig = await getOidcConfig(config.portaUrl, org);
    
    const currentUrl = new URL(ctx.href);
    const tokenResponse = await exchangeCode(
      oidcConfig, currentUrl, state, nonce, code_verifier
    );

    // Store tokens in session
    ctx.session!.tokens = {
      access_token: tokenResponse.access_token,
      id_token: tokenResponse.id_token,
      refresh_token: tokenResponse.refresh_token,
      token_type: tokenResponse.token_type,
      expires_at: tokenResponse.expires_in
        ? Date.now() + tokenResponse.expires_in * 1000
        : undefined,
    };
    ctx.session!.orgKey = orgKey;
    delete ctx.session!.oidc; // Clean up PKCE state

    ctx.redirect('/');
  });

  // POST /auth/logout
  router.post('/auth/logout', async (ctx) => {
    const orgKey = ctx.session!.orgKey || 'no2fa';
    const idToken = ctx.session!.tokens?.id_token;
    const org = config.organizations[orgKey];

    ctx.session = null; // Destroy session

    if (idToken && org) {
      const oidcConfig = await getOidcConfig(config.portaUrl, org);
      const endSessionUrl = getEndSessionUrl(oidcConfig, idToken, config.bffUrl);
      ctx.redirect(endSessionUrl.toString());
    } else {
      ctx.redirect('/');
    }
  });
}
```

### JWT Decode Helper (`src/helpers/jwt.ts`)

```typescript
/**
 * Decode a JWT for display purposes (no verification — display only).
 */
export function decodeJwt(token: string): { header: object; payload: object } | null {
  try {
    const parts = token.split('.');
    if (parts.length !== 3) return null;
    const header = JSON.parse(Buffer.from(parts[0], 'base64url').toString());
    const payload = JSON.parse(Buffer.from(parts[1], 'base64url').toString());
    return { header, payload };
  } catch {
    return null;
  }
}
```

## Integration Points

### With Porta
- Discovery: `GET /{orgSlug}/.well-known/openid-configuration`
- Authorization: `GET /{orgSlug}/auth`
- Token: `POST /{orgSlug}/token`
- UserInfo: `GET /{orgSlug}/me`
- Introspection: `POST /{orgSlug}/token/introspection`
- End Session: `GET /{orgSlug}/session/end`

### With Redis
- Session store: `bff:sess:{sessionId}` keys
- Same Redis instance as Porta (port 6379), different key prefix

### With Seed Script
- Reads `config.generated.json` at startup
- Depends on BFF clients + M2M client being seeded first

## Error Handling

| Error Case | Handling Strategy |
| --- | --- |
| Config file missing | Fail fast with clear error message |
| Discovery fails | Log error, show on dashboard as "Porta unavailable" |
| Token exchange fails | Log error, redirect to dashboard with error flash |
| Session expired | Redirect to login |
| Redis unavailable | Fallback to memory session (dev convenience) |
| Invalid state/nonce | Return 400 with explanation |
