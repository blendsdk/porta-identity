# Playground Application: Vanilla HTML/JS SPA

> **Document**: 05-playground-app.md
> **Parent**: [Index](00-index.md)

## Overview

A zero-build-step vanilla HTML/JavaScript SPA served by sirv-cli on port 4000.
Uses vendored `oidc-client-ts` for OIDC Authorization Code + PKCE flows. Provides
a scenario selector, configuration panel, token dashboard, UserInfo panel, and
status indicators for interactive testing of all Porta auth flows.

## Architecture

### File Structure

```
playground/
├── package.json              # sirv-cli dev dependency, start script
├── index.html                # Main page: scenario selector + dashboard
├── callback.html             # OIDC redirect callback (minimal page)
├── css/
│   └── style.css             # All styles, CSS variables for theming
├── js/
│   ├── app.js                # Main entry: init, scenario selector, event wiring
│   ├── auth.js               # OIDC UserManager setup, login/logout/callback
│   ├── config.js             # Config loading from config.generated.js, org/client presets
│   ├── tokens.js             # JWT decode (base64url), token display, refresh
│   ├── userinfo.js           # UserInfo endpoint fetch and display
│   └── ui.js                 # DOM helpers, status indicators, theme toggle, event log
├── vendor/
│   └── oidc-client-ts.min.js # Vendored oidc-client-ts browser bundle
├── config.generated.js       # Auto-generated by seed (gitignored)
└── README.md                 # Usage instructions
```

### Module Dependencies

```
config.generated.js (data)
        ↓
    config.js (loads config, provides presets)
        ↓
    auth.js (creates UserManager with config)
        ↓ ↑
    app.js (orchestrates everything)
      ├── tokens.js (decodes/displays tokens from auth)
      ├── userinfo.js (fetches /userinfo using access token)
      └── ui.js (DOM updates, status checks, theme)
```

All modules use ES module `import/export` syntax with `<script type="module">`.

### `js/auth.js` — OIDC Client

```javascript
/**
 * OIDC authentication module.
 * Wraps oidc-client-ts UserManager for login, logout, callback, and refresh.
 */
import { UserManager, WebStorageStateStore } from '/vendor/oidc-client-ts.min.js';
import { logEvent } from '/js/ui.js';

let userManager = null;
let currentUser = null;

/**
 * Initialize the OIDC UserManager with the given settings.
 * @param {{ authority: string, clientId: string }} settings
 */
export function initAuth(settings) {
  userManager = new UserManager({
    authority: settings.authority,
    client_id: settings.clientId,
    redirect_uri: 'http://localhost:4000/callback.html',
    post_logout_redirect_uri: 'http://localhost:4000/',
    response_type: 'code',
    scope: 'openid profile email',
    automaticSilentRenew: false,
    monitorSession: false,
    userStore: new WebStorageStateStore({ store: window.sessionStorage }),
  });

  // Wire up events for the event log
  userManager.events.addUserLoaded((user) => {
    currentUser = user;
    logEvent('user_loaded', 'User loaded — tokens received');
  });
  userManager.events.addUserUnloaded(() => {
    currentUser = null;
    logEvent('user_unloaded', 'User unloaded — session cleared');
  });
  userManager.events.addAccessTokenExpiring(() => {
    logEvent('token_expiring', 'Access token expiring soon');
  });

  logEvent('auth_init', `OIDC client initialized — authority: ${settings.authority}`);
  return userManager;
}

/** Start the OIDC authorization flow (redirect to Porta login page) */
export async function login() {
  if (!userManager) throw new Error('Auth not initialized');
  logEvent('login_start', 'Starting OIDC authorization request...');
  await userManager.signinRedirect();
}

/** Handle the OIDC callback (exchange code for tokens) */
export async function handleCallback() {
  // Create a temporary UserManager to process the callback
  const tempManager = new UserManager({
    authority: sessionStorage.getItem('pg_authority') || 'http://localhost:3000/playground-no2fa',
    client_id: sessionStorage.getItem('pg_client_id') || '',
    redirect_uri: 'http://localhost:4000/callback.html',
    response_type: 'code',
    scope: 'openid profile email',
    userStore: new WebStorageStateStore({ store: window.sessionStorage }),
  });
  try {
    const user = await tempManager.signinRedirectCallback();
    window.location.href = '/';
  } catch (err) {
    document.getElementById('callback-message').textContent =
      `Error: ${err.message}. Redirecting...`;
    setTimeout(() => { window.location.href = '/'; }, 3000);
  }
}

/** Get the current authenticated user (or null) */
export async function getUser() {
  if (!userManager) return null;
  currentUser = await userManager.getUser();
  return currentUser;
}

/** Perform RP-initiated logout */
export async function logout() {
  if (!userManager) return;
  logEvent('logout_start', 'Initiating RP-initiated logout...');
  await userManager.signoutRedirect();
}

/** Manually trigger a silent token refresh */
export async function refreshToken() {
  if (!userManager) return null;
  logEvent('refresh_start', 'Triggering silent token refresh...');
  try {
    currentUser = await userManager.signinSilent();
    logEvent('refresh_success', 'Token refresh successful');
    return currentUser;
  } catch (err) {
    logEvent('refresh_error', `Token refresh failed: ${err.message}`);
    throw err;
  }
}

/** Clear all auth state without server-side logout */
export async function clearState() {
  if (!userManager) return;
  await userManager.removeUser();
  currentUser = null;
}

/**
 * Store current OIDC settings in sessionStorage so callback.html can
 * reconstruct the UserManager.
 */
export function storeAuthSettings(settings) {
  sessionStorage.setItem('pg_authority', settings.authority);
  sessionStorage.setItem('pg_client_id', settings.clientId);
}
```

### `js/tokens.js` — Token Decoding & Display

```javascript
/**
 * Token decoding and display utilities.
 * Decodes JWT (ID tokens) using base64url without external libraries.
 */

/**
 * Decode a JWT payload (no signature verification — display only).
 * @param {string} jwt - The JWT string
 * @returns {{ header: object, payload: object }}
 */
export function decodeJwt(jwt) {
  const parts = jwt.split('.');
  if (parts.length !== 3) throw new Error('Invalid JWT format');
  return {
    header: JSON.parse(base64UrlDecode(parts[0])),
    payload: JSON.parse(base64UrlDecode(parts[1])),
  };
}

/** Base64URL decode to string */
function base64UrlDecode(str) {
  // Replace base64url chars with base64 chars, add padding
  let base64 = str.replace(/-/g, '+').replace(/_/g, '/');
  while (base64.length % 4) base64 += '=';
  return atob(base64);
}

/**
 * Render all token panels in the dashboard.
 * @param {object} user - The oidc-client-ts User object
 */
export function displayTokens(user) {
  displayIdToken(user);
  displayAccessToken(user);
  displayRefreshToken(user);
}

/** Render the ID token panel with decoded header + payload */
function displayIdToken(user) {
  const panel = document.getElementById('id-token-panel');
  if (!user.id_token) {
    panel.innerHTML = '<h3>ID Token</h3><p class="muted">Not present</p>';
    return;
  }
  const decoded = decodeJwt(user.id_token);
  const expiresAt = new Date(decoded.payload.exp * 1000);
  panel.innerHTML = `
    <h3>ID Token</h3>
    <div class="token-section">
      <h4>Header</h4>
      <pre>${JSON.stringify(decoded.header, null, 2)}</pre>
    </div>
    <div class="token-section">
      <h4>Payload (Claims)</h4>
      <pre>${JSON.stringify(decoded.payload, null, 2)}</pre>
    </div>
    <div class="token-meta">
      <span>Expires: ${expiresAt.toLocaleString()}</span>
      <span>Subject: ${decoded.payload.sub}</span>
    </div>
  `;
}

/** Render the access token panel (opaque — show raw + metadata) */
function displayAccessToken(user) {
  const panel = document.getElementById('access-token-panel');
  if (!user.access_token) {
    panel.innerHTML = '<h3>Access Token</h3><p class="muted">Not present</p>';
    return;
  }
  const expiresAt = user.expires_at ? new Date(user.expires_at * 1000) : null;
  const scopes = user.scope || 'N/A';
  // Try to decode if it's a JWT; otherwise show as opaque
  let tokenContent;
  try {
    const decoded = decodeJwt(user.access_token);
    tokenContent = `<pre>${JSON.stringify(decoded.payload, null, 2)}</pre>`;
  } catch {
    tokenContent = `<pre class="raw-token">${user.access_token}</pre>`;
  }
  panel.innerHTML = `
    <h3>Access Token</h3>
    ${tokenContent}
    <div class="token-meta">
      ${expiresAt ? `<span>Expires: ${expiresAt.toLocaleString()}</span>` : ''}
      <span>Scopes: ${scopes}</span>
    </div>
  `;
}

/** Render the refresh token panel (presence indicator + metadata) */
function displayRefreshToken(user) {
  const panel = document.getElementById('refresh-token-panel');
  if (!user.refresh_token) {
    panel.innerHTML = '<h3>Refresh Token</h3><p class="muted">Not present</p>';
    return;
  }
  panel.innerHTML = `
    <h3>Refresh Token</h3>
    <p class="success">✅ Refresh token received</p>
    <pre class="raw-token">${user.refresh_token.substring(0, 20)}...</pre>
  `;
}

/** Clear all token panels */
export function clearTokenDisplay() {
  ['id-token-panel', 'access-token-panel', 'refresh-token-panel'].forEach(id => {
    document.getElementById(id).innerHTML = '';
  });
}
```

### `js/userinfo.js` — UserInfo Endpoint

```javascript
/**
 * UserInfo endpoint integration.
 * Fetches the OIDC UserInfo endpoint and displays the response.
 */
import { logEvent } from '/js/ui.js';

/**
 * Fetch the UserInfo endpoint using the current access token.
 * @param {string} accessToken - The access token
 * @param {string} authority - The OIDC authority URL
 */
export async function fetchUserInfo(accessToken, authority) {
  logEvent('userinfo_start', 'Fetching UserInfo endpoint...');
  const discoveryUrl = `${authority}/.well-known/openid-configuration`;

  // Get the userinfo_endpoint from discovery
  const discoveryRes = await fetch(discoveryUrl);
  const discovery = await discoveryRes.json();
  const userinfoUrl = discovery.userinfo_endpoint;

  // Call userinfo with Bearer token
  const res = await fetch(userinfoUrl, {
    headers: { Authorization: `Bearer ${accessToken}` },
  });

  if (!res.ok) {
    const errText = await res.text();
    logEvent('userinfo_error', `UserInfo failed: ${res.status} ${errText}`);
    throw new Error(`UserInfo request failed: ${res.status}`);
  }

  const data = await res.json();
  logEvent('userinfo_success', 'UserInfo response received');
  return data;
}

/**
 * Display UserInfo response in the UI panel.
 * Highlights RBAC roles and custom claims separately.
 */
export function displayUserInfo(data) {
  const panel = document.getElementById('userinfo-panel');
  const json = document.getElementById('userinfo-json');
  panel.hidden = false;
  json.textContent = JSON.stringify(data, null, 2);
}

/** Hide the UserInfo panel */
export function hideUserInfo() {
  document.getElementById('userinfo-panel').hidden = true;
}
```

```javascript
/**

/**
 * Check if a URL is reachable (simple fetch with timeout).
 * Used for status indicators.
 */
async function isReachable(url, timeoutMs = 3000) {
  try {
    const controller = new AbortController();
    const timeout = setTimeout(() => controller.abort(), timeoutMs);
    const res = await fetch(url, { signal: controller.signal, mode: 'no-cors' });
    clearTimeout(timeout);
    return true;
  } catch {
    return false;
  }
}

/** Update status indicator dots */
export async function updateStatusIndicators(config, isLoggedIn) {
  // Porta health check
  const portaOk = config ? await isReachable(`${config.portaUrl}/health`) : false;
  setStatusDot('status-porta', portaOk, portaOk ? 'Porta: Connected' : 'Porta: Unreachable');

  // MailHog check
  const mailhogOk = config ? await isReachable(config.mailhogUrl) : false;
  setStatusDot('status-mailhog', mailhogOk, mailhogOk ? 'MailHog: Connected' : 'MailHog: Unreachable');

  // Auth state
  setStatusDot('status-auth', isLoggedIn, isLoggedIn ? 'Logged In' : 'Logged Out');
}

/** Set a status dot to green (ok) or red (not ok) */
function setStatusDot(elementId, ok, tooltip) {
  const el = document.getElementById(elementId);
  if (!el) return;
  el.classList.toggle('status-ok', ok);
  el.classList.toggle('status-error', !ok);
  el.title = tooltip;
}

/**
 * Add an event to the event log.
 * @param {string} type - Event type (for CSS class)
 * @param {string} message - Human-readable message
 */
export function logEvent(type, message) {
  const events = document.getElementById('events');
  if (!events) return;
  const entry = document.createElement('div');
  entry.className = `event event-${type}`;
  const time = new Date().toLocaleTimeString();
  entry.innerHTML = `<span class="event-time">${time}</span> <span class="event-msg">${message}</span>`;
  events.prepend(entry); // Newest at top
  // Keep max 50 entries
  while (events.children.length > 50) events.lastChild.remove();
}

/** Clear the event log */
export function clearEventLog() {
  const events = document.getElementById('events');
  if (events) events.innerHTML = '';
}

/** Initialize dark/light theme toggle */
export function initThemeToggle() {
  const btn = document.getElementById('theme-toggle');
  if (!btn) return;
  // Default to dark
  const saved = localStorage.getItem('pg-theme') || 'dark';
  document.documentElement.setAttribute('data-theme', saved);
  btn.textContent = saved === 'dark' ? '☀️' : '🌙';
  btn.addEventListener('click', () => {
    const current = document.documentElement.getAttribute('data-theme');
    const next = current === 'dark' ? 'light' : 'dark';
    document.documentElement.setAttribute('data-theme', next);
    localStorage.setItem('pg-theme', next);
    btn.textContent = next === 'dark' ? '☀️' : '🌙';
  });
}
```

### `js/app.js` — Main Application

```javascript
/**
 * Main application entry point.
 * Initializes config, renders scenarios, wires up buttons, manages state.
 */
import { loadConfig, getConfig, getOrgSettings, SCENARIOS } from '/js/config.js';
import { initAuth, login, logout, getUser, refreshToken, clearState, storeAuthSettings } from '/js/auth.js';
import { displayTokens, clearTokenDisplay } from '/js/tokens.js';
import { fetchUserInfo, displayUserInfo, hideUserInfo } from '/js/userinfo.js';
import { updateStatusIndicators, logEvent, initThemeToggle, setAuthView } from '/js/ui.js';

let currentScenario = null;

/** Application initialization */
async function init() {
  logEvent('init', 'Playground starting...');
  initThemeToggle();

  // Load config
  const config = await loadConfig();
  if (!config) {
    logEvent('error', 'Failed to load config.generated.js — run seed first');
    return;
  }
  logEvent('config_loaded', 'Configuration loaded');

  // Render scenario buttons
  renderScenarios();

  // Populate org selector
  populateOrgSelector(config);

  // Check for existing session
  const defaultOrg = getOrgSettings('no2fa');
  initAuth(defaultOrg);
  const user = await getUser();
  if (user && !user.expired) {
    logEvent('session_restored', `Existing session found for ${user.profile.email || user.profile.sub}`);
    displayTokens(user);
    setAuthView(true);
  } else {
    setAuthView(false);
  }

  // Update status indicators
  await updateStatusIndicators(config, !!user && !user.expired);

  // Wire up buttons
  document.getElementById('btn-login')?.addEventListener('click', handleLogin);
  document.getElementById('btn-logout')?.addEventListener('click', handleLogout);
  document.getElementById('btn-relogin')?.addEventListener('click', handleRelogin);
  document.getElementById('btn-userinfo')?.addEventListener('click', handleUserInfo);
  document.getElementById('btn-refresh')?.addEventListener('click', handleRefresh);
  document.getElementById('config-org')?.addEventListener('change', handleOrgChange);

  logEvent('ready', 'Playground ready');
}

/** Render scenario buttons in the sidebar */
function renderScenarios() {
  const container = document.getElementById('scenarios');
  if (!container) return;
  // Keep the h2
  const h2 = container.querySelector('h2');
  container.innerHTML = '';
  if (h2) container.appendChild(h2);

  SCENARIOS.forEach(scenario => {
    const btn = document.createElement('button');
    btn.className = 'scenario-btn';
    btn.dataset.scenarioId = scenario.id;
    btn.innerHTML = `
      <strong>${scenario.name}</strong>
      <small>${scenario.description}</small>
    `;
    btn.addEventListener('click', () => selectScenario(scenario));
    container.appendChild(btn);
  });
}

/** Select a scenario: configure OIDC client and show instructions */
function selectScenario(scenario) {
  currentScenario = scenario;
  const config = getConfig();
  const orgSettings = getOrgSettings(scenario.orgKey);
  const userInfo = config.scenarios[scenario.id];

  // Update org selector to match
  const orgSelect = document.getElementById('config-org');
  if (orgSelect) orgSelect.value = scenario.orgKey;

  // Re-initialize auth with new org
  initAuth(orgSettings);
  storeAuthSettings(orgSettings);

  // Update config details display
  updateConfigDetails(orgSettings);

  // Highlight active scenario
  document.querySelectorAll('.scenario-btn').forEach(btn => {
    btn.classList.toggle('active', btn.dataset.scenarioId === scenario.id);
  });

  // Log with test user hint
  const email = userInfo?.userEmail || 'see sidebar';
  const user = config.users[email];
  logEvent('scenario', `Selected: ${scenario.name}`);
  if (scenario.hint) {
    logEvent('hint', `💡 ${scenario.hint}`);
  }
  if (user) {
    logEvent('credentials', `📧 ${email} / ${user.password}`);
  }
}

// ... additional handler functions (handleLogin, handleLogout, etc.)
// Each handler is a thin wrapper calling the auth/tokens/userinfo modules
```

## Error Handling

| Error Case | Handling Strategy |
|------------|-------------------|
| `config.generated.js` missing | Show error banner with "Run seed first" message |
| OIDC discovery unreachable | Status dot turns red; login button shows error |
| Token exchange fails | Callback page shows error, redirects to home after 3s |
| UserInfo call fails | Show error in userinfo panel, log to event log |
| Token refresh fails | Log error, suggest re-login |
